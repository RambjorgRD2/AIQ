```javascript
import { supabase } from '../lib/supabase.ts'

const form = document.getElementById('signin-form') as HTMLFormElement
const emailInput = document.getElementById('email-input') as HTMLInputElement
const submitBtn = form.querySelector('button[type="submit"]') as HTMLButtonElement
const formState = document.getElementById('form-state')!
const successState = document.getElementById('success-state')!
const errorState = document.getElementById('error-state')!
const sentEmail = document.getElementById('sent-email')!
const errorMessage = document.getElementById('error-message')!

function showState(state: 'form' | 'success' | 'error') {
  formState.style.display = state === 'form' ? 'block' : 'none'
  successState.style.display = state === 'success' ? 'block' : 'none'
  errorState.style.display = state === 'error' ? 'block' : 'none'
}

form.addEventListener('submit', async (e) => {
  e.preventDefault()
  const email = emailInput.value.trim()
  if (!email) return

  submitBtn.disabled = true
  submitBtn.textContent = 'Sending...'

  // Store intended redirect for after callback
  sessionStorage.setItem('auth_redirect', document.referrer || '/AIQ/')

  const { error } = await supabase.auth.signInWithOtp({
    email,
    options: {
      emailRedirectTo: 'https://RambjorgRD2.github.io/AIQ/auth/callback',
      shouldCreateUser: true,
    },
  })

  if (error) {
    errorMessage.textContent = error.message
    showState('error')
  } else {
    sentEmail.textContent = email
    showState('success')
  }

  submitBtn.disabled = false
  submitBtn.textContent = 'Send Magic Link'
})

// "Try again" and "Back to sign in" reset to form
document.querySelectorAll('[data-reset-form]').forEach(el => {
  el.addEventListener('click', () => showState('form'))
})
```

**Client-side `<script>` (NOT `is:inline`):**
```javascript
import { supabase } from '../../lib/supabase.ts'

// Supabase client auto-detects #access_token on init (detectSessionInUrl: true)
// Wait for the SIGNED_IN event, then redirect
supabase.auth.onAuthStateChange((event, session) => {
  if (event === 'SIGNED_IN' && session) {
    const redirectTo = sessionStorage.getItem('auth_redirect') || '/AIQ/'
    sessionStorage.removeItem('auth_redirect')
    window.location.href = redirectTo
  }
})

// Fallback: if already signed in (e.g., page refresh after auth)
supabase.auth.getSession().then(({ data: { session } }) => {
  if (session) {
    const redirectTo = sessionStorage.getItem('auth_redirect') || '/AIQ/'
    sessionStorage.removeItem('auth_redirect')
    window.location.href = redirectTo
  }
})

// Timeout fallback: if nothing happens after 10 seconds, show manual link
setTimeout(() => {
  const fallback = document.getElementById('callback-fallback')
  if (fallback) fallback.style.display = 'block'
}, 10000)
```

**Key details:**
- The page directory `src/pages/auth/` must be created (it doesn't exist yet)
- `detectSessionInUrl: true` (configured in supabase.ts) means the Supabase client automatically parses the `#access_token=...&refresh_token=...` hash fragment from the URL
- The `onAuthStateChange` listener fires when the session is established
- The fallback `getSession()` call handles the case where the session was already parsed before the listener was registered
- The 10-second timeout shows a manual fallback link
