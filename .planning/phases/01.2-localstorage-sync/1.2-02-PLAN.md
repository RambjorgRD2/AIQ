    Find the existing onAuthStateChange listener. In the `SIGNED_IN` event handler (or add one if only SIGNED_OUT is handled), add a migration function that runs once per sign-in:

    ```javascript
    async function migrateLocalStorageToSupabase(userId) {
      const { upsertLessonProgress, upsertAssessmentResult, upsertReflectionNote } = await import('/src/lib/db.ts');

      // Migrate lesson completions
      const courseMap = [
        { key: 'aiq-foundations-completed', courseId: 'foundations' },
        { key: 'aiq-workflows-completed', courseId: 'workflows' },
        { key: 'aiq-leadership-completed', courseId: 'leadership' },
      ];
      for (const { key, courseId } of courseMap) {
        const ids = JSON.parse(localStorage.getItem(key) || '[]');
        for (const lessonId of ids) {
          await upsertLessonProgress({ userId, lessonId, courseId }).catch(console.error);
        }
      }

      // Migrate reflection notes — scan all localStorage keys
      const prefixes = [
        { prefix: 'reflection-', courseId: 'foundations' },
        { prefix: 'wf-reflection-', courseId: 'workflows' },
        { prefix: 'ld-reflection-', courseId: 'leadership' },
      ];
      for (let i = 0; i < localStorage.length; i++) {
        const k = localStorage.key(i);
        if (!k) continue;
        for (const { prefix } of prefixes) {
          if (k.startsWith(prefix)) {
            const lessonId = k.slice(prefix.length);
            const noteText = localStorage.getItem(k) || '';
            if (noteText) {
              await upsertReflectionNote({ userId, lessonId, noteText }).catch(console.error);
            }
          }
        }
      }

      // Migrate latest assessment result
      const raw = localStorage.getItem('aiq_latest_result');
      if (raw) {
        try {
          const r = JSON.parse(raw);
          await upsertAssessmentResult({
            userId,
            score: r.score,
            tier: r.tier,
            tierLabel: r.tierLabel || r.tier_label || '',
            radarData: r.radarData || r.radar_data,
          }).catch(console.error);
        } catch (e) { /* malformed JSON, skip */ }
      }
    }
    ```

    Call `migrateLocalStorageToSupabase(session.user.id)` inside the SIGNED_IN handler. This is fire-and-forget — do not await in a way that blocks the auth flow.
