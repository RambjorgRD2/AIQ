    ```javascript
    async function loadResult() {
      // First priority: fresh result from this session (just completed assessment)
      const fresh = localStorage.getItem('aiq_latest_result');
      if (fresh) {
        try {
          result = JSON.parse(fresh);
          renderResults(result); // whatever the existing render function is called
          return;
        } catch (e) { /* malformed, fall through */ }
      }

      // Second priority: persisted result from Supabase (authenticated users returning)
      const { data: { session } } = await supabase.auth.getSession();
      if (session?.user) {
        const { getLatestAssessmentResult } = await import('/src/lib/db.ts');
        const stored = await getLatestAssessmentResult(session.user.id).catch(() => null);
        if (stored) {
          // Show a "last score" banner (not a fresh result)
          result = {
            score: stored.score,
            tier: stored.tier,
            tierLabel: stored.tier_label,
            radarData: stored.radar_data,
            _isStoredResult: true,  // flag so the UI can show the "Your last AIQ score" banner
          };
          renderResults(result);
          return;
        }
      }

      // No result at all — show the "no result" state that already exists
      renderResults(null);
    }
    loadResult();
    ```
